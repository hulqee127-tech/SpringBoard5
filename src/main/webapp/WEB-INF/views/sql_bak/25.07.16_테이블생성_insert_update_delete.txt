num int primary key auto_increment,
title varchar(300) not null,

);

create table board(
    num NUMBER(6) PRIMARY KEY, 
    title VARCHAR2(300) NOT NULL,
    content clob ,    
    writer varchar2(60),
    wdate DATE DEFAULT SYSDATE,
    hit number(6)
);


create sequence seq_num;
